HarmonyOS ArkTS开发入门与架构设计

HarmonyOS ArkTS ArkUI 声明式UI

在智旅修记项目中，我使用HarmonyOS NEXT和ArkTS开发了原生鸿蒙应用。作为华为推出的新一代操作系统，HarmonyOS提供了独特的技术栈和开发范式。本文将分享ArkTS开发的核心知识和实践经验。

为什么选择HarmonyOS？

HarmonyOS NEXT(纯血鸿蒙)是华为推出的全场景分布式操作系统，具有以下特点：

  • 原生性能：不再依赖Android兼容层，性能更优
  • 一次开发，多端部署：支持手机、平板、车机、智能穿戴等多设备
  • 分布式能力：跨设备协同、流转、共享
  • ArkTS语言：基于TypeScript，前端开发者上手快

ArkTS核心特性

ArkTS是鸿蒙应用开发的主力语言，在TypeScript基础上进行了扩展：

ArkTS增强特性：
  • 声明式UI：使用装饰器语法定义组件和状态
  • 状态管理：@State、@Prop、@Link、@Provide等装饰器
  • 并发模型：TaskPool和Worker多线程方案
  • 类型安全：严格的类型检查，编译期发现问题

状态管理详解

状态管理是ArkTS开发的核心，理解各种装饰器的使用场景至关重要：

@State - 组件内部状态

@Component
struct Counter {
  @State count: number = 0
  
  build() {
    Column() {
      Text(`计数: ${this.count}`)
        .fontSize(20)
      
      Button('增加')
        .onClick(() => {
          this.count++  // 状态变化自动触发UI更新
        })
    }
  }
}

@Prop - 父子组件单向同步

// 父组件
@Entry
@Component
struct Parent {
  @State parentCount: number = 10
  
  build() {
    Column() {
      Child({ childCount: this.parentCount })
    }
  }
}

// 子组件
@Component
struct Child {
  @Prop childCount: number  // 从父组件接收，单向同步
  
  build() {
    Button(`子组件: ${this.childCount}`)
      .onClick(() => {
        this.childCount++  // 只影响子组件
      })
  }
}

@Link - 父子组件双向同步

@Component
struct TwoWayBinding {
  @Link value: number
  
  build() {
    Slider({
      value: $$this.value,  // 双向绑定语法
      min: 0,
      max: 100
    })
  }
}

// 使用
TwoWayBinding({ value: this.$parentValue })

2. 图片修复服务实现

// services/ImageService.ets
import { http } from '@kit.NetworkKit'

export class ImageService {
  private baseUrl: string = 'https://api.example.com/image'
  
  // 图像去雾
  async dehaze(imageUri: string): Promise<string> {
    const response = await http.createHttp().request(
      `${this.baseUrl}/dehaze`,
      {
        method: http.RequestMethod.POST,
        header: { 'Content-Type': 'application/json' },
        extraData: { image: imageUri }
      }
    )
    
    if (response.responseCode === 200) {
      const result = JSON.parse(response.result as string)
      return result.processedImageUrl
    }
    throw new Error('去雾处理失败')
  }
  
  // 超分辨率
  async superResolution(imageUri: string, scale: number = 2): Promise<string> {
    const response = await http.createHttp().request(
      `${this.baseUrl}/sr`,
      {
        method: http.RequestMethod.POST,
        extraData: { 
          image: imageUri,
          scale: scale  // 2x 或 4x
        }
      }
    )
    
    return JSON.parse(response.result as string).resultUrl
  }
}

3. 页面状态管理

// pages/ImageFix.ets
import { ImageService } from '../services/ImageService'

@Entry
@Component
struct ImageFix {
  @State selectedImage: string = ''
  @State isProcessing: boolean = false
  @State resultImage: string = ''
  @State currentMode: FixMode = FixMode.DEHAZE
  
  private imageService: ImageService = new ImageService()
  
  enum FixMode {
    DEHAZE,
    SUPER_RESOLUTION
  }
  
  build() {
    Column({ space: 16 }) {
      // 图片选择区域
      ImageSelector({
        imageUri: this.selectedImage,
        onImageSelected: (uri) => this.selectedImage = uri
      })
      
      // 模式选择
      Row({ space: 12 }) {
        ModeButton({
          text: '去雾',
          isSelected: this.currentMode === FixMode.DEHAZE,
          onClick: () => this.currentMode = FixMode.DEHAZE
        })
        
        ModeButton({
          text: '超分辨率',
          isSelected: this.currentMode === FixMode.SUPER_RESOLUTION,
          onClick: () => this.currentMode = FixMode.SUPER_RESOLUTION
        })
      }
      
      // 处理按钮
      Button(this.isProcessing ? '处理中...' : '开始修复')
        .enabled(!this.isProcessing && this.selectedImage !== '')
        .onClick(() => this.processImage())
      
      // 结果展示
      if (this.resultImage !== '') {
        Image(this.resultImage)
          .width('100%')
          .height(300)
          .objectFit(ImageFit.Contain)
      }
    }
    .padding(16)
  }
  
  async processImage() {
    this.isProcessing = true
    try {
      if (this.currentMode === FixMode.DEHAZE) {
        this.resultImage = await this.imageService.dehaze(this.selectedImage)
      } else {
        this.resultImage = await this.imageService.superResolution(this.selectedImage)
      }
    } catch (error) {
      promptAction.showToast({ message: '处理失败: ' + error.message })
    } finally {
      this.isProcessing = false
    }
  }
}

数据持久化

使用AppStorage和LocalStorage实现数据持久化：

// 全局状态存储
import { AppStorageV2 } from '@kit.ArkUI'

// 定义旅行计划数据类
@ObservedV2
class TravelPlan {
  @Trace id: number
  @Trace destination: string
  @Trace startDate: Date
  @Trace endDate: Date
  @Trace budget: number
  
  constructor(id: number, destination: string) {
    this.id = id
    this.destination = destination
    this.startDate = new Date()
    this.endDate = new Date()
    this.budget = 0
  }
}

// 存储管理类
class StorageManager {
  static readonly PLAN_KEY = 'travel_plans'
  
  // 保存计划
  static savePlans(plans: TravelPlan[]) {
    AppStorageV2.set(StorageManager.PLAN_KEY, plans)
    // 同步到本地存储
    const context = getContext()
    const path = context.filesDir + '/plans.json'
    fs.write(path, JSON.stringify(plans))
  }
  
  // 读取计划
  static getPlans(): TravelPlan[] {
    return AppStorageV2.get(StorageManager.PLAN_KEY) || []
  }
}

后台任务处理

图片处理等耗时操作需要在后台执行，避免阻塞UI：

import { taskpool } from '@kit.ArkTS'

// 定义后台任务
@Concurrent
function processImageInBackground(imageData: ArrayBuffer): ArrayBuffer {
  // 在子线程执行图像处理算法
  const processed = applyDehazeAlgorithm(imageData)
  return processed
}

// 调用后台任务
async function processImage(imageUri: string) {
  const imageData = await loadImageData(imageUri)
  
  // 提交到任务池
  const task = new taskpool.Task(processImageInBackground, imageData)
  const result = await taskpool.execute(task)
  
  return saveProcessedImage(result)
}
