迭代式加盐慢哈希 - 密码安全存储最佳实践

作者：高俊杰 2026-03-08 10 分钟阅读

什么是密码哈希？

密码哈希是将用户密码转换成一串不可逆的字符串的过程。即使数据库被泄露，攻击者也无法直接获取用户的原始密码。

为什么需要加盐？

问题：如果两个用户使用相同的密码，他们的哈希值也会相同

解决：为每个密码添加唯一的随机值（盐），即使密码相同，哈希值也不同

为什么需要慢哈希？

问题：快速哈希算法（如 MD5、SHA-1）让攻击者可以快速尝试大量密码

解决：使用计算密集型的慢哈希算法，增加攻击者的成本

Argon2 算法简介

Argon2 是目前最推荐的密码哈希算法，它是 Password Hashing Competition（PHC）的获胜者。

Python 实现

from argon2 import PasswordHasher
from argon2.exceptions import VerifyMismatchError

class PasswordManager:
    def __init__(self):
        self.ph = PasswordHasher(
            time_cost=3,
            memory_cost=65536,
            parallelism=4,
            hash_len=32,
            salt_len=16
        )
    
    def hash_password(self, password: str) -> str:
        return self.ph.hash(password)
    
    def verify_password(self, hash_str: str, password: str) -> bool:
        try:
            self.ph.verify(hash_str, password)
            return True
        except VerifyMismatchError:
            return False

if __name__ == "__main__":
    pm = PasswordManager()
    
    original_pwd = "123456"
    print(f"原始密码：{original_pwd}")
    
    hash1 = pm.hash_password(original_pwd)
    print(f"哈希结果 1: {hash1}")
    
    hash2 = pm.hash_password(original_pwd)
    print(f"哈希结果 2: {hash2}")
    print(f"两次哈希是否不同：{hash1 != hash2}")
    
    result1 = pm.verify_password(hash1, original_pwd)
    print(f"验证正确密码：{result1}")
    
    result2 = pm.verify_password(hash1, "wrongpassword")
    print(f"验证错误密码：{result2}")

C++ Qt 实现

#include <QCoreApplication>
#include <QDebug>
#include <QCryptographicHash>
#include <QRandomGenerator>
#include <QTimer>

class PasswordManager {
public:
    static const int SALT_LENGTH = 16;
    static const int ITERATIONS = 100000;
    
    static QByteArray generateSalt() {
        QByteArray salt;
        salt.resize(SALT_LENGTH);
        QRandomGenerator::system()->fillRange(
            reinterpret_cast<uchar*>(salt.data()), SALT_LENGTH);
        return salt;
    }
    
    static QByteArray hashPassword(const QString& password, const QByteArray& salt) {
        QByteArray data = password.toUtf8() + salt;
        
        for (int i = 0; i < ITERATIONS; ++i) {
            data = QCryptographicHash::hash(data, QCryptographicHash::Sha256);
        }
        
        return data.toHex();
    }
    
    static bool verifyPassword(const QString& password, const QByteArray& salt, const QByteArray& hash) {
        QByteArray computedHash = hashPassword(password, salt);
        return computedHash == hash;
    }
};

int main(int argc, char *argv[]) {
    QCoreApplication a(argc, argv);
    
    QString password = "123456";
    qDebug() << "原始密码:" << password;
    
    QByteArray salt1 = PasswordManager::generateSalt();
    QByteArray hash1 = PasswordManager::hashPassword(password, salt1);
    qDebug() << "哈希结果 1:" << hash1;
    
    QByteArray salt2 = PasswordManager::generateSalt();
    QByteArray hash2 = PasswordManager::hashPassword(password, salt2);
    qDebug() << "哈希结果 2:" << hash2;
    qDebug() << "两次哈希是否不同:" << (hash1 != hash2);
    
    bool result1 = PasswordManager::verifyPassword(password, salt1, hash1);
    qDebug() << "验证正确密码:" << result1;
    
    bool result2 = PasswordManager::verifyPassword("wrongpassword", salt1, hash1);
    qDebug() << "验证错误密码:" << result2;
    
    return a.exec();
}
