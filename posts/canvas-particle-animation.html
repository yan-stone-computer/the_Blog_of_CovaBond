Canvas粒子动画与赛博朋克视觉设计

Canvas JavaScript 粒子系统 赛博朋克

在竞语星河项目中，我使用Canvas实现了一套赛博朋克风格的粒子动画系统。本文将分享粒子动画的核心原理、性能优化技巧，以及如何打造沉浸式的赛博朋克视觉体验。

赛博朋克视觉风格

赛博朋克(Cyberpunk)是一种科幻美学风格，具有以下视觉特征：

  • 霓虹配色：高饱和度的粉色、青色、紫色
  • 深色背景：黑色或深灰色为主色调
  • 网格线条：透视网格营造空间感
  • 故障效果：Glitch艺术、扫描线
  • 发光元素：霓虹光晕、发光边框

粒子系统基础

粒子系统是计算机图形学中模拟模糊现象的技术，如火焰、烟雾、星光等。

粒子类设计

class Particle {
  constructor(x, y, options = {}) {
    this.x = x;
    this.y = y;
    this.vx = (Math.random() - 0.5) * options.speed || 1;
    this.vy = (Math.random() - 0.5) * options.speed || 1;
    this.radius = options.radius || Math.random() * 3 + 1;
    this.color = options.color || '#FF2D78';
    this.life = 1; // 生命值 1-0
    this.decay = options.decay || 0.01;
    this.glow = options.glow || false;
  }

  update() {
    this.x += this.vx;
    this.y += this.vy;
    this.life -= this.decay;
    
    // 边界反弹
    if (this.x < 0 || this.x > canvas.width) this.vx *= -1;
    if (this.y < 0 || this.y > canvas.height) this.vy *= -1;
  }

  draw(ctx) {
    ctx.save();
    ctx.globalAlpha = this.life;
    
    // 发光效果
    if (this.glow) {
      ctx.shadowBlur = 15;
      ctx.shadowColor = this.color;
    }
    
    ctx.beginPath();
    ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2);
    ctx.fillStyle = this.color;
    ctx.fill();
    ctx.restore();
  }

  isDead() {
    return this.life <= 0;
  }
}

粒子管理器

class ParticleSystem {
  constructor(canvas) {
    this.canvas = canvas;
    this.ctx = canvas.getContext('2d');
    this.particles = [];
    this.maxParticles = 200; // 性能控制
  }

  emit(x, y, count = 1, options = {}) {
    for (let i = 0; i < count; i++) {
      if (this.particles.length >= this.maxParticles) {
        this.particles.shift(); // 移除最老的粒子
      }
      this.particles.push(new Particle(x, y, options));
    }
  }

  update() {
    this.particles = this.particles.filter(p => {
      p.update();
      return !p.isDead();
    });
  }

  draw() {
    this.ctx.clearRect(0, 0, this.canvas.width, this.canvas.height);
    this.particles.forEach(p => p.draw(this.ctx));
  }

  animate() {
    this.update();
    this.draw();
    requestAnimationFrame(() => this.animate());
  }
}

力导向图算法

在竞语星河中，节点之间通过引力相互连接，形成动态网络：

class ForceDirectedGraph {
  constructor(nodes, links) {
    this.nodes = nodes.map(n => ({
      ...n,
      vx: 0,
      vy: 0,
      x: Math.random() * width,
      y: Math.random() * height
    }));
    this.links = links;
  }

  update() {
    // 1. 节点间斥力
    for (let i = 0; i < this.nodes.length; i++) {
      for (let j = i + 1; j < this.nodes.length; j++) {
        const dx = this.nodes[j].x - this.nodes[i].x;
        const dy = this.nodes[j].y - this.nodes[i].y;
        const dist = Math.sqrt(dx * dx + dy * dy);
        
        if (dist > 0 && dist < 200) {
          const force = 1000 / (dist * dist);
          const fx = (dx / dist) * force;
          const fy = (dy / dist) * force;
          
          this.nodes[i].vx -= fx;
          this.nodes[i].vy -= fy;
          this.nodes[j].vx += fx;
          this.nodes[j].vy += fy;
        }
      }
    }

    // 2. 连线引力
    this.links.forEach(link => {
      const source = this.nodes[link.source];
      const target = this.nodes[link.target];
      
      const dx = target.x - source.x;
      const dy = target.y - source.y;
      const dist = Math.sqrt(dx * dx + dy * dy);
      
      const force = (dist - 100) * 0.01; // 目标距离100px
      const fx = (dx / dist) * force;
      const fy = (dy / dist) * force;
      
      source.vx += fx;
      source.vy += fy;
      target.vx -= fx;
      target.vy -= fy;
    });

    // 3. 中心引力（防止飞散）
    this.nodes.forEach(node => {
      const dx = centerX - node.x;
      const dy = centerY - node.y;
      node.vx += dx * 0.0001;
      node.vy += dy * 0.0001;
      
      // 速度衰减
      node.vx *= 0.9;
      node.vy *= 0.9;
      
      // 更新位置
      node.x += node.vx;
      node.y += node.vy;
    });
  }

  draw(ctx) {
    // 绘制连线
    this.links.forEach(link => {
      const source = this.nodes[link.source];
      const target = this.nodes[link.target];
      
      ctx.beginPath();
      ctx.moveTo(source.x, source.y);
      ctx.lineTo(target.x, target.y);
      ctx.strokeStyle = 'rgba(0, 240, 255, 0.2)';
      ctx.lineWidth = 1;
      ctx.stroke();
    });

    // 绘制节点
    this.nodes.forEach(node => {
      ctx.beginPath();
      ctx.arc(node.x, node.y, node.radius || 5, 0, Math.PI * 2);
      ctx.fillStyle = node.color || '#00F0FF';
      ctx.shadowBlur = 20;
      ctx.shadowColor = node.color || '#00F0FF';
      ctx.fill();
    });
  }
}

触摸交互效果

添加触摸交互，让粒子跟随手指移动：

// 触摸粒子效果
class TouchParticleEffect {
  constructor(canvas) {
    this.canvas = canvas;
    this.ctx = canvas.getContext('2d');
    this.particles = [];
    this.touchX = null;
    this.touchY = null;
    
    this.bindEvents();
  }

  bindEvents() {
    this.canvas.addEventListener('touchmove', (e) => {
      e.preventDefault();
      const touch = e.touches[0];
      this.touchX = touch.clientX;
      this.touchY = touch.clientY;
      this.emit(5);
    });

    this.canvas.addEventListener('mousemove', (e) => {
      this.touchX = e.clientX;
      this.touchY = e.clientY;
      this.emit(2);
    });
  }

  emit(count) {
    for (let i = 0; i < count; i++) {
      this.particles.push({
        x: this.touchX + (Math.random() - 0.5) * 20,
        y: this.touchY + (Math.random() - 0.5) * 20,
        vx: (Math.random() - 0.5) * 2,
        vy: (Math.random() - 0.5) * 2,
        life: 1,
        color: ['#FF2D78', '#00F0FF', '#F0C040'][Math.floor(Math.random() * 3)]
      });
    }
  }

  update() {
    this.particles = this.particles.filter(p => {
      p.x += p.vx;
      p.y += p.vy;
      p.life -= 0.02;
      return p.life > 0;
    });
  }

  draw() {
    this.particles.forEach(p => {
      this.ctx.save();
      this.ctx.globalAlpha = p.life;
      this.ctx.shadowBlur = 10;
      this.ctx.shadowColor = p.color;
      this.ctx.fillStyle = p.color;
      this.ctx.beginPath();
      this.ctx.arc(p.x, p.y, 3, 0, Math.PI * 2);
      this.ctx.fill();
      this.ctx.restore();
    });
  }
}

性能优化技巧

Canvas性能优化要点：
  • 粒子数量控制：移动端不超过200个，桌面端不超过500个
  • 分层渲染：背景、粒子、UI分层，减少重绘区域
  • 离屏Canvas：静态元素预先渲染到离屏Canvas
  • requestAnimationFrame：使用RAF代替setInterval
  • 对象池：复用粒子对象，减少GC压力

对象池模式

class ParticlePool {
  constructor(size) {
    this.pool = [];
    this.active = [];
    
    // 预创建粒子
    for (let i = 0; i < size; i++) {
      this.pool.push(new Particle());
    }
  }

  get() {
    if (this.pool.length > 0) {
      const particle = this.pool.pop();
      this.active.push(particle);
      return particle;
    }
    return null; // 池已满
  }

  release(particle) {
    const index = this.active.indexOf(particle);
    if (index > -1) {
      this.active.splice(index, 1);
      this.pool.push(particle);
    }
  }

  reset() {
    this.pool.push(...this.active);
    this.active.length = 0;
  }
}

function drawScanlines(ctx, width, height) {
  ctx.save();
  ctx.fillStyle = 'rgba(0, 0, 0, 0.1)';
  
  for (let y = 0; y < height; y += 4) {
    ctx.fillRect(0, y, width, 2);
  }
  
  ctx.restore();
}

霓虹发光文字

function drawNeonText(ctx, text, x, y) {
  ctx.save();
  
  // 外发光
  ctx.shadowColor = '#FF2D78';
  ctx.shadowBlur = 30;
  ctx.fillStyle = '#FF2D78';
  ctx.font = 'bold 48px Arial';
  ctx.fillText(text, x, y);
  
  // 内发光
  ctx.shadowBlur = 10;
  ctx.fillStyle = '#FFFFFF';
  ctx.fillText(text, x, y);
  
  ctx.restore();
}

总结

Canvas粒子动画可以创造出极具视觉冲击力的效果，但需要注意性能优化。在实际项目中，要根据目标设备合理控制粒子数量，使用对象池减少内存分配，并采用分层渲染策略。赛博朋克风格的视觉设计需要精心调配色彩和光影，才能营造出沉浸式的未来感体验。
