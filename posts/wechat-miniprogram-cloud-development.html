微信小程序云开发实战：多端协同架构设计

微信小程序 云开发 云函数 多端协同

在慧码验修和竞语星河项目中，我使用微信小程序云开发构建了多端协同的应用系统。云开发让开发者无需搭建服务器即可实现数据存储、云函数、文件存储等功能。本文将分享云开发的架构设计和实战经验。

为什么选择云开发？

传统的小程序开发需要自行搭建后端服务器、数据库、域名备案等，而云开发提供了开箱即用的解决方案：

  • 零服务器运维：无需购买服务器，自动弹性扩缩容
  • 免费额度充足：个人开发者完全够用
  • 一体化开发：数据库、存储、云函数统一管理
  • 天然鉴权：集成微信登录，用户身份自动验证

慧码验修架构设计

慧码验修是一个三端协同的工单管理系统，包含用户端、维修工端、管理员端。

系统架构：
  • 用户端：提交报修、查看订单状态、评价反馈
  • 维修工端：接单处理、收入统计、工单管理
  • 管理员端：数据看板、人员管理、公告发布
  • 统一后端：云数据库 + 云函数 + 云存储

云函数实现

// cloudfunctions/createOrder/index.js
const cloud = require('wx-server-sdk')
cloud.init()

exports.main = async (event, context) => {
  const { description, type, address, images } = event
  const { OPENID } = cloud.getWXContext()
  
  const db = cloud.database()
  
  // 生成订单号
  const orderNo = 'WX' + Date.now()
  
  // 创建订单
  const result = await db.collection('orders').add({
    data: {
      orderNo,
      userId: OPENID,
      status: 'pending',
      type,
      description,
      images,
      address,
      timeline: [{
        status: 'created',
        time: db.serverDate(),
        note: '订单创建成功'
      }],
      createTime: db.serverDate(),
      updateTime: db.serverDate()
    }
  })
  
  // 发送订阅消息
  await sendSubscribeMessage(OPENID, 'ORDER_CREATED', {
    orderNo,
    type,
    createTime: new Date().toLocaleString()
  })
  
  return {
    code: 0,
    data: { orderId: result._id, orderNo }
  }
}

多端权限控制

三端使用同一套云函数，通过角色字段进行权限控制：

// 权限中间件
async function checkRole(requiredRoles) {
  const { OPENID } = cloud.getWXContext()
  const db = cloud.database()
  
  const user = await db.collection('users').doc(OPENID).get()
  
  if (!user.data || !requiredRoles.includes(user.data.role)) {
    throw new Error('权限不足')
  }
  
  return user.data
}

// 维修工接单
exports.acceptOrder = async (event) => {
  // 验证维修工身份
  const worker = await checkRole(['worker', 'admin'])
  
  const { orderId } = event
  const db = cloud.database()
  
  // 原子性更新：只能接待接单状态的订单
  const result = await db.collection('orders').where({
    _id: orderId,
    status: 'pending'
  }).update({
    data: {
      workerId: worker._openid,
      status: 'accepted',
      updateTime: db.serverDate(),
      timeline: db.command.push({
        status: 'accepted',
        time: db.serverDate(),
        note: `维修工${worker.name}已接单`
      })
    }
  })
  
  if (result.stats.updated === 0) {
    return { code: -1, message: '订单已被其他维修工接单' }
  }
  
  return { code: 0, message: '接单成功' }
}

智能派单算法

系统根据距离和评分自动推荐最优维修工：

// 智能派单
exports.findBestWorker = async (event) => {
  const { address, type } = event
  const db = cloud.database()
  const _ = db.command
  
  // 1. 获取在线且擅长该类型的维修工
  const workers = await db.collection('users').where({
    role: 'worker',
    status: 'online',
    skills: _.in([type]),
    currentOrders: _.lt(5) // 当前订单少于5个
  }).get()
  
  // 2. 计算综合得分
  const scoredWorkers = workers.data.map(worker => {
    // 距离得分（越近越高）
    const distance = calculateDistance(address, worker.location)
    const distanceScore = Math.max(0, 100 - distance * 10)
    
    // 评分得分
    const ratingScore = (worker.rating || 5) * 10
    
    // 响应速度得分
    const responseScore = worker.avgResponseTime < 30 ? 20 : 10
    
    // 综合得分
    const totalScore = distanceScore * 0.4 + ratingScore * 0.4 + responseScore * 0.2
    
    return { ...worker, distance, totalScore }
  })
  
  // 3. 按得分排序
  scoredWorkers.sort((a, b) => b.totalScore - a.totalScore)
  
  return {
    code: 0,
    data: scoredWorkers.slice(0, 3) // 返回前3名
  }
}

实时通知推送

使用微信订阅消息实现订单状态变更通知：

// 发送订阅消息
async function sendSubscribeMessage(openid, templateId, data) {
  try {
    await cloud.openapi.subscribeMessage.send({
      touser: openid,
      templateId: getTemplateId(templateId),
      data: {
        character_string1: { value: data.orderNo },
        thing2: { value: data.type },
        time3: { value: data.time },
        thing4: { value: data.status }
      }
    })
  } catch (err) {
    console.error('订阅消息发送失败:', err)
  }
}

// 订单状态变更时触发
exports.updateOrderStatus = async (event) => {
  const { orderId, status, note } = event
  const db = cloud.database()
  
  // 更新订单状态
  const order = await db.collection('orders').doc(orderId).get()
  await db.collection('orders').doc(orderId).update({
    data: {
      status,
      updateTime: db.serverDate(),
      timeline: db.command.push({
        status,
        time: db.serverDate(),
        note
      })
    }
  })
  
  // 通知用户
  await sendSubscribeMessage(order.data.userId, 'STATUS_CHANGE', {
    orderNo: order.data.orderNo,
    type: order.data.type,
    time: new Date().toLocaleString(),
    status: getStatusText(status)
  })
  
  return { code: 0 }
}

数据可视化

管理员端使用ECharts展示统计数据：

// 获取统计数据
exports.getStatistics = async (event) => {
  const { startDate, endDate } = event
  const db = cloud.database()
  const _ = db.command
  
  // 订单统计
  const orderStats = await db.collection('orders').where({
    createTime: _.gte(new Date(startDate)).and(_.lte(new Date(endDate)))
  }).get()
  
  // 按状态分组
  const statusCount = {}
  const dailyCount = {}
  
  orderStats.data.forEach(order => {
    // 状态统计
    statusCount[order.status] = (statusCount[order.status] || 0) + 1
    
    // 每日统计
    const date = order.createTime.toISOString().split('T')[0]
    dailyCount[date] = (dailyCount[date] || 0) + 1
  })
  
  // 收入统计
  const income = orderStats.data
    .filter(o => o.status === 'completed')
    .reduce((sum, o) => sum + (o.price || 0), 0)
  
  return {
    code: 0,
    data: {
      totalOrders: orderStats.data.length,
      statusCount,
      dailyCount,
      totalIncome: income,
      avgRating: calculateAvgRating(orderStats.data)
    }
  }
}

性能优化技巧

1. 数据库索引

为常用查询字段建立索引：

  • userId（查询用户订单）
  • status（按状态筛选）
  • createTime（时间范围查询）
  • workerId（维修工订单）

2. 分页查询

// 分页获取订单列表
exports.getOrders = async (event) => {
  const { page = 1, pageSize = 20, status } = event
  const db = cloud.database()
  
  let query = db.collection('orders')
  if (status) {
    query = query.where({ status })
  }
  
  const result = await query
    .orderBy('createTime', 'desc')
    .skip((page - 1) * pageSize)
    .limit(pageSize)
    .get()
  
  return { code: 0, data: result.data }
}

3. 云函数并发

使用Promise.all批量处理：
